如果是一次只能擦两个，希望求进行几次呢。一个（为了介绍$lowbit$）想法是，拆成若干个$2^{p_i}$。
这里我们需要取出$n$（二进制）最低一位。这是一个常见的操作（$lowbit$）。基于补码机制和按位与可以快速实现$lowbit$操作。

int lowbit(int x){
	return x&-x;
}

对于每次$n$个元素即为一颗满二叉树的叶子节点。可得共$2n-1$个节点，$n-1$个非叶子节点。
完整代码：

#include<iostream>
#include<map>
using namespace std;
int lowbit(int x){
	return x&-x;
}
int main(){
	int T;
	cin>>T;
	while(T--){
		int n,ans=0;
		cin>>n;
		while(n){
			ans+=lowbit(n)-1;
			n-=lowbit(n);
		}
		cout<<ans<<endl;
	}
	return 0;
}

C

题意：给定一序列$a$，$P(i)=(\sum_i^n a_i>=x)$成立，求关于$i$的最值。
运用前缀和和二分可使时间复杂度降到可接受的程度。
实现时注意数据范围

#include<iostream>
#include<algorithm>
using namespace std;

typedef long long LL;
const int N=2*1e5+10;
LL n,x;
LL a[N],s[N];

bool check(int i){
	if(s[n]-s[i-1]>=x) return true;
	return false;
}

int search(){
	int l=1,r=n,ans=-1;
	while(l<=r){
		int mid=(l+r)>>1;
		if(check(mid)){
			ans=mid;
			l=mid+1;
		}
		else r=mid-1;
	}
	return ans;
}

int main(){
	int T;
	cin>>n>>T;
	for(int i=1;i<=n;++i)
		cin>>a[i];
	sort(a+1,a+n+1);
	for(int i=1;i<=n;++i)
		s[i]=s[i-1]+a[i];
	while(T--){
		cin>>x;
		int res=search();
		if(res==-1) puts("-1");
		else cout<<n-res+1<<endl;
	}
	return 0;		
}

我们常用二分法来处理一个具有性质两段性序列，求解他的分段处的索引。在设计算法时，一定要明确所求索引那一段的性质到底是什么。
下面这种二分的写法相较其他模板多了$ans$变量，但是由于处理的都是闭区间，又少了很多细节：

int search(){
	int l=1,r=n,ans=-1;
	while(l<=r){
		int mid=(l+r)>>1;
		if(check(mid)){
			ans=mid;
			l=mid+1;
		}
		else r=mid-1;
	}
	return ans;
}

题意：给定始末坐标。对每一个时间$t$，我们都有确定的$a_t$和不确定的$b_t$.
存在$c_{begin}+\sum (a_t+b_t)=c_{dest}$
求$t_{min}$.

我们不仅要研究决策空间，还要研究目标值域！

先计算风向，再把曼哈顿距离和天数比较即可判定。

#include<iostream>
using namespace std;
typedef long long LL;
int n;
LL x_b,y_b,x_d,y_d;
string s;
int cnt[512]; 
bool check(LL t){
	LL x=x_b,y=y_b;
	for(LL i=0;i<t%n;++i){
		if(s[i]=='U') ++y;
		if(s[i]=='D') --y;
		if(s[i]=='L') --x;
		if(s[i]=='R') ++x;
	}
	y+=(cnt['U']-cnt['D'])*(t/n);
	x+=(cnt['R']-cnt['L'])*(t/n);
	if(t>=abs(x-x_d)+abs(y-y_d)) return true;
	return false;
}
LL search(){
	LL l=1,r=2*1e14,ans=-1;
	while(l<=r){
		LL mid=(l+r)>>1;
//		printf("%d:\n",mid);
		if(check(mid)){
			ans=mid;
			r=mid-1;
		}
		else l=mid+1;
	}
	return ans;
}
int main()
{
	cin>>x_b>>y_b>>x_d>>y_d>>n>>s;
	for(int i=0;i<s.size();++i)
		cnt[s[i]]++;
	cout<<search();
	return 0;
}

H

注意观察每一位对答案的贡献，不难发现$a_i$的贡献仅和$b_j(i<=j<=n)$有关。
类似高精度，实现时可翻转容器

#include<iostream>
#include<algorithm>
using namespace std;
const int N=2*1e5+10;
const int MOD=998244353;
/*
998244353=(114514+114514)*((1+1)*4*514+((11+4)*(5-1)*4+1-14+5+14))+(114514+(114*51*4+(11*45*(1+4)+1*14+5*14)))
*/
typedef long long LL;
LL n,m;
string a,b;
int s[N];
LL work(){
	LL res=0;
	LL p=1;
	for(int i=0;i<n;++i){
		res+=(a[i]-'0')*p*s[i];
		res%=MOD;
		p*=2;
		p%=MOD;
	}
	return res;
}
int main(){
	cin>>n>>m>>a>>b;
	reverse(a.begin(),a.end());
	reverse(b.begin(),b.end());
	for(int i=b.size()-1;i>=0;--i){
		if(i==b.size()-1) s[i]=b[i]-'0';
		else s[i]=s[i+1]+b[i]-'0';
	}
	cout<<work();
	return 0;
}
